factory OrderWire.trigger({
  required int asset,
  required bool isBuy,
  required String limitPx,
  required String sz,
  required String triggerPx,
  required TpslType tpsl,
  bool isMarket = true,
  bool reduceOnly = true,
}) {
  return OrderWire(
    asset: asset,
    isBuy: isBuy,
    limitPx: limitPx,
    sz: sz,
    reduceOnly: reduceOnly,
    orderType: {
      'trigger': {
        'isMarket': isMarket,
        'triggerPx': triggerPx,
        'tpsl': tpsl.value,
      }
    },
  );
}
